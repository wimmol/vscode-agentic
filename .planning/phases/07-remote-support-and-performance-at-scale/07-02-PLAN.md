       ```typescript

       ```

       Add a new `updateDiffStatusForAgent` method:
       ```typescript
       async updateDiffStatusForAgent(repoPath: string, agentName: string): Promise<void> {
         if (!this.diffService) return;
         const key = `${repoPath}::${agentName}`;
         const lastChecked = this.diffTimestamps.get(key) ?? 0;
         if (Date.now() - lastChecked < this.DIFF_TTL_MS) return; // TTL still valid

         const hasDiffs = await this.diffService.hasUnmergedChanges(repoPath, agentName);
         this.diffStatusCache.set(key, hasDiffs);
         this.diffTimestamps.set(key, Date.now());
         this.refresh();
       }
       ```

       Refactor existing `updateDiffStatus()` to iterate all agents but delegate to `updateDiffStatusForAgent` (respects TTL):
       ```typescript
       async updateDiffStatus(): Promise<void> {
         if (!this.diffService) return;
         const agents = this.agentService.getAll();
         for (const agent of agents) {
           await this.updateDiffStatusForAgent(agent.repoPath, agent.agentName);
         }
       }
       ```

       Add `invalidateDiffCache` method for explicit cache busting (useful when agent is deleted or status changes significantly):
       ```typescript
       invalidateDiffCache(repoPath: string, agentName: string): void {
         const key = `${repoPath}::${agentName}`;
         this.diffStatusCache.delete(key);
         this.diffTimestamps.delete(key);
       }
       ```

       Update `dispose()` to clear the timestamp map:
       ```typescript
       dispose(): void {
         // ... existing cleanup ...
         this.diffTimestamps.clear();
       }
       ```

       **Important:** The `debouncedDiffUpdate()` method (called on every `onDidChangeAgents` event) should continue calling `updateDiffStatus()` which now respects TTL. This means the 300ms debounce + 30s TTL together prevent thundering herds: debounce collapses rapid events, TTL skips agents checked recently.

       Add a `handleAgentLimitError` function (in the same file, not exported -- private helper):
       ```typescript
       async function handleAgentLimitError(
         error: AgentLimitError,
         agentService: AgentService,
       ): Promise<boolean> {
         // Find oldest idle agent (not running, not suspended) to offer for suspension
         const candidates = error.existingAgents
           .filter(a => a.status !== "running" && a.status !== "suspended")
           .sort((a, b) => a.createdAt.localeCompare(b.createdAt));

         if (candidates.length === 0) {
           vscode.window.showWarningMessage(
             `${error.message} No idle agents available to suspend.`,
           );
           return false;
         }

         const oldest = candidates[0];
         const scopeLabel = error.limitType === "per-repo"
           ? `Per-repo agent limit (${error.limit})`
           : `Global agent limit (${error.limit})`;

         const action = await vscode.window.showWarningMessage(
           `${scopeLabel} reached. Suspend idle agent '${oldest.agentName}' to make room?`,
           "Suspend & Create",
           "Cancel",
         );

         if (action === "Suspend & Create") {
           await agentService.suspendAgent(oldest.repoPath, oldest.agentName);
           return true;
         }
         return false;
       }
       ```
