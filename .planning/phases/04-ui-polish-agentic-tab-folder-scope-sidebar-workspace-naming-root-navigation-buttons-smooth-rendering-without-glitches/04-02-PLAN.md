    ```typescript
    refresh(): void {
      if (!this._view) return;
      if (!this._initialRenderDone) {
        // Webview not yet resolved or first render not done -- skip postMessage
        return;
      }
      const data = this._buildDashboardData();
      this._view.webview.postMessage({ type: "update", data });
    }
    ```

    ```typescript
    resolveWebviewView(...): void {
      this._view = webviewView;

      this._initialRenderDone = true;
      // Message handling...
    }
    ```

    6. Add `_buildDashboardData()`:
    ```typescript
    private _buildDashboardData(): { repos: Array<{path: string, name: string, agents: AgentEntry[]}>, scope: string } {
      const repos = this.repoConfigService.getAll();
      return {
        repos: repos.map(repo => ({
          path: repo.path,
          name: path.basename(repo.path),
          agents: this.agentService.getForRepo(repo.path),
        })),
        scope: this._currentScope,
      };
    }
    ```
